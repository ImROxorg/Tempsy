-- Scripts:
local function closeScript()
    close.MouseButton1Click:Connect(function()
        local frame = close.Parent.Parent.Parent
        frame:TweenPosition(UDim2.new(0.4, 0, 1.5, 0))
        infjumpenabled = false
        wait(1)
        frame.Parent:Destroy()
    end)
end
closeScript()

local function infiniteJumpScript()
    InfiniteJump.MouseButton1Click:Connect(function()
        if infjumpenabled == false then
            infjumpenabled = true
            InfiniteJump.Text = "Enabled"
        elseif infjumpenabled == true then
            infjumpenabled = false
            InfiniteJump.Text = "Disabled"
        end
    end)
end
infiniteJumpScript()

local function draggableScript()
    Main.Selectable = true
    Main.Active = true
    Main.Draggable = true
end
draggableScript()

local function openingScript()
    Main.Position = UDim2.new(0.33, 0, 2, 0)
    Main:TweenPosition(UDim2.new(0.33, 0, 0.3, 0))
end
openingScript()

local function infJumpScript()
    game:GetService("UserInputService").JumpRequest:Connect(function()
        if infjumpenabled == true then
            game.Players.LocalPlayer.Character:FindFirstChild("Humanoid"):ChangeState("Jumping")
        end
    end)
end
infJumpScript()
